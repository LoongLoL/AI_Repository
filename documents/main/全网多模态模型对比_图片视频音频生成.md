## 五、OpenClaw 最佳配置方案

### 5.1 全免费配置（推荐）

```json5
{
  agents: {
    defaults: {
      // === 主模型（纯文本对话）===
      model: {
        primary: "openrouter/owl-alpha",
        fallbacks: [
          "openrouter/qwen/qwen3-coder:free",
          "deepseek/deepseek-v4-flash"
        ]
      },

      // === 图片/多模态理解 ===
      imageModel: {
        primary: "google/gemini-2.5-flash",          // 免费，1M上下文，全模态
        fallbacks: [
          "openrouter/qwen/qwen3-vl-235b-a22b-instruct",  // Qwen3 VL 旗舰
          "openrouter/nvidia/nemotron-3-nano-omni-30b-a3b-reasoning:free"  // 免费备选
        ]
      },

      // === 图片生成 ===
      imageGenerationModel: {
        primary: "google/gemini-3.1-flash-image-preview",  // 输入免费，输出极便宜
        fallbacks: [
          "openai/gpt-5-image-mini"                          // 极便宜备选
        ]
      },

      // === 视频生成 ===
      videoGenerationModel: {
        primary: "fal/bytedance/seedance-2.0/fast/text-to-video",  // fal.ai，按量付费
        fallbacks: [
          "runway/gen4.5"                                           // Runway，质量最好
        ]
      },

      // === 音乐/音频生成 ===
      musicGenerationModel: {
        primary: "fal/fal-ai/minimax-music/v2.6",  // fal.ai
        fallbacks: [
          "google/lyria-3-clip-preview"              // OpenRouter，$0.04/clip
        ]
      }
    }
  }
}
```
